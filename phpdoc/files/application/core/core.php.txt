/**
 * Ядро создает автолоадеры
 *
 * @author fiftystars
 */
class core {
    private static $instance = NULL;
    
    public function __construct() {
        
        //автолоадер для классов
        spl_autoload_register(function($className){
            $fileName = "application/core/classes/{$className}.class.php";
            if (file_exists($fileName)){
                require_once $fileName;
            }
        });
        // автолоадер для трейтов
        spl_autoload_register(function($traitName){
            $fileName = "application/core/traits/{$traitName}.trait.php";
            if (file_exists($fileName)){
                require_once $fileName;
            }
        });
        // автолоадер для интерфейсов
        spl_autoload_register(function($interfaceName){
            $fileName = "application/core/interfaces/{$interfaceName}.interface.php";
            if (file_exists($fileName)){
                require_once $fileName;
            }
        });
        
        self::$instance = $this;
    }
    
    
    // PENDING необходим ли объект или можно обойтись static?
    public static function getInstance() {
        if (self::$instance === NULL){
           new self();
        }
        return self::$instance;
    }
}
